[docs]    def run(self):
        event_handler = Handler()
        event_handler.watch = self.watch
        self.observer.schedule(event_handler, self.watch, recursive = True)
        self.observer.start()
        try:
            while True:
                time.sleep(5)
        except:
            self.observer.stop()
            print("Observer Stopped")
        self.observer.join()
